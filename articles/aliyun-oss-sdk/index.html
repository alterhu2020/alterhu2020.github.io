重构源码¶

class AliyunAuth(object):
    '''
    阿里云oss参考api
    '''
    def __init__(self,ak=aliyun_access_key,sk=aliyun_secret_key,bucketname=aliyun_bucket_name,defaultdomain=aliyun_default_domain):
        self.__ak=ak
        self.__sk=sk
        self.__bucketname=bucketname
        self.__defaultdomain=defaultdomain
        self.__auth=oss2.Auth(ak,sk)
        self.__bucket=oss2.Bucket(self.__auth, defaultdomain, bucketname)

    def upload(self,filestream,prefix,filename):
        '''
        上传文件,
        :param filestream: 第一个参数可以是字符串也可以是字节数组
        :param prefix: 上传的目录前缀
        :param filename:  上传的文件名称
        :return:  无
        '''
        key=prefix+filename
        self.__bucket.put_object(key, filestream)
